/**

 * @extends \Gobl\ORM\ORMResults<\Gobl\Tests\Db\Client>

abstract class ClientsResults extends \Gobl\ORM\ORMResults
{
	/**
	 * ClientsResults constructor.
	 */
	public function __construct(\Gobl\DBAL\Queries\QBSelect $query)
	{
		parent::__construct(
			\Gobl\Tests\Db\Client::TABLE_NAMESPACE,
			\Gobl\Tests\Db\Client::TABLE_NAME,
			$query
		);

	}

	/**
	 * {@inheritDoc}
	 * 
	 * @return static
	 */

	{
		return new ClientsResultsReal($query);
	}
}
